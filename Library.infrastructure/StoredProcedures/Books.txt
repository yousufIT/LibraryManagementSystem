-- الإجراء المخزن للحصول على جميع الكتب
CREATE PROCEDURE GetAllBooks
AS
BEGIN
    SET NOCOUNT ON;
    SELECT * FROM Books;
END
GO

-- الإجراء المخزن للحصول على كتاب بواسطة المعرف
CREATE PROCEDURE GetBookById
    @ID INT
AS
BEGIN
    SET NOCOUNT ON;
    SELECT * FROM Books WHERE BookID = @ID;
END
GO

-- الإجراء المخزن لإنشاء كتاب جديد
CREATE PROCEDURE CreateBook
    @Title NVARCHAR(100),
    @PublishedDate DATE,
    @Description NVARCHAR(MAX),
    @MainCategoryID INT,
    @SubCategoryID INT,
    @Publisher NVARCHAR(100),
    @Pages INT,
    @Language NVARCHAR(50),
    @Price DECIMAL(18, 2),
    @Edition NVARCHAR(50),
    @Format NVARCHAR(50),
    @Location NVARCHAR(100),
    @BookID INT OUTPUT
AS
BEGIN
    SET NOCOUNT ON;
    BEGIN TRANSACTION;
    
    INSERT INTO Books (Title, PublishedDate, Description, MainCategoryID, SubCategoryID, Publisher, Pages, Language, Price, Edition, Format, Location)
    VALUES (@Title, @PublishedDate, @Description, @MainCategoryID, @SubCategoryID, @Publisher, @Pages, @Language, @Price, @Edition, @Format, @Location);
    
    SET @BookID = SCOPE_IDENTITY();
    
    SELECT * FROM Books WHERE BookID = @BookID;
    COMMIT TRANSACTION;
END
GO

-- الإجراء المخزن لتحديث كتاب موجود
CREATE PROCEDURE UpdateBook
    @ID INT,
    @Title NVARCHAR(100),
    @PublishedDate DATE,
    @Description NVARCHAR(MAX),
    @MainCategoryID INT,
    @SubCategoryID INT,
    @Publisher NVARCHAR(100),
    @Pages INT,
    @Language NVARCHAR(50),
    @Price DECIMAL(18, 2),
    @Edition NVARCHAR(50),
    @Format NVARCHAR(50),
    @Location NVARCHAR(100)
AS
BEGIN
    SET NOCOUNT ON;
    BEGIN TRANSACTION;
    
    UPDATE Books
    SET Title = @Title,
        PublishedDate = @PublishedDate,
        Description = @Description,
        MainCategoryID = @MainCategoryID,
        SubCategoryID = @SubCategoryID,
        Publisher = @Publisher,
        Pages = @Pages,
        Language = @Language,
        Price = @Price,
        Edition = @Edition,
        Format = @Format,
        Location = @Location
    WHERE BookID = @ID;
    
    COMMIT TRANSACTION;
END
GO

-- الإجراء المخزن لحذف كتاب
CREATE PROCEDURE DeleteBook
    @ID INT
AS
BEGIN
    SET NOCOUNT ON;
    BEGIN TRANSACTION;
    
    DELETE FROM Books WHERE BookID = @ID;
    
    COMMIT TRANSACTION;
END
GO

-- الإجراء المخزن للحصول على الكتب التي كتبها مؤلف معين
CREATE PROCEDURE GetBooksByAuthor
    @AuthorID INT
AS
BEGIN
    SET NOCOUNT ON;
    SELECT b.* 
    FROM Books b
    INNER JOIN BookAuthors ba ON b.BookID = ba.BookID
    WHERE ba.AuthorID = @AuthorID;
END
GO
